Обучение GIT

Первые команды

git help

git init

git status

git add "file_name or ."

git commit -m "massage"

git log

git show com_HASH

git diff

git checkout file_name

-откат до последнего коммита

git checkout com_HASH file_name

- откат но конретного коммита.

git diff --staged

git reset HEAD file_name

- сбросить один файл из инедексации перед коммитом

git commit --amend -m "massage"

-изменения в сообщении комммита

git rm file_name

-удаление файла

git rm --cached file_name

-удаление файла из комита без удаления из папки

git commit --amend -no-edit

git checkout -b yellow-design com_HASH

- откат на коммит с созданием метки на него

git merge yellow-design -m "massage"
